/* Minimal Slider */ var content, step, count, contentWidth, maxPosition, left, right, set, currentPosition = 0, currentStep = 0, move = false, currentStop = 0; function rightm() { hide(); currentPosition -= 10; // step; for only step work upadte(); if (currentPosition <= currentStop ) { clearInterval(set); move = false; }; hide(); } function leftm() { hide(); currentPosition += 10; upadte(); if (currentPosition >= currentStop ) { clearInterval(set); move = false; }; hide(); }; function init() { content = document.getElementById("content"); mrg = parseInt(content.style.marginLeft); step = document.getElementsByTagName('li')[0].offsetWidth; count = document.getElementById("content").childNodes.length; contentWidth = step * count; maxPosition = - (contentWidth - step); left = document.getElementById("left"); right = document.getElementById("right"); content.style.width = contentWidth + 'px'; right.addEventListener("click", function(){ // rightm(); hide(); currentStep++; for only step work if (currentStep>=0&&move==false) { currentStep++; currentStop = -(currentStep*step); set = setInterval( 'rightm()', 10 ); } }); left.addEventListener("click", function(){ if (currentStep>=0&&move==false) { currentStep--; currentStop = -(currentStep*step); set = setInterval( 'leftm()', 10); } }); hide(); return content, maxPosition, currentPosition, move; }; function upadte() { content.style.marginLeft = currentPosition + 'px'; move = true; //console.log( currentPosition, currentStep, currentStop, move ); } function hide(){ if (parseInt(content.style.marginLeft) >= 0) { left.style.display = "none"; right.style.display = "block" } else if (parseInt(content.style.marginLeft) <= maxPosition){ right.style.display = "none"; left.style.display = "block" } else { right.style.display = "block"; left.style.display = "block" } }; window.onload = init;
